Var
nome, telefone, email : caractere
id_gerado : inteiro
sucesso_gravacao : logico

Inicio

// Entrada de Dados
Escreva("Informe o nome do cliente: ")
Leia(nome)

Escreva("Informe o telefone: ")
Leia(telefone)

Escreva("Informe o e-mail: ")
Leia(email)


// Processamento e Validação
Se (nome = "") entao
      Escreval("Erro: O nome do cliente é obrigatório.")
Senao

      // Simulação do ID gerado automaticamente no banco
      id_gerado <- 1  

      // Simulação da execução do SQL INSERT
      // No sistema real, isso ocorreria em salvar-cliente.php
      sucesso_gravacao <- Verdadeiro

      Se (sucesso_gravacao = Verdadeiro) entao
            Escreval("Sucesso: Cliente registrado no banco de dados.")
            Escreval("Código gerado automaticamente: ", id_gerado)
            Escreval("Redirecionando para listar-cliente.php...")
      Senao
            Escreval("Falha: Erro ao conectar ao banco de dados.")
      FimSe

FimSe
